37# Importing data from OECD
38consumption_data = pd.read_csv('consumption_data.csv')
39gdp_data = pd.read_csv('gdp_data.csv')
40
41# Data preprocessing
42consumption_data = consumption_data[['LOCATION', 'Value']]
43gdp_data = gdp_data[['LOCATION', 'Value']]
44
45data = pd.merge(consumption_data, gdp_data, on='LOCATION', suffixes=('_consumption', '_gdp'))
46data = data.dropna()
47data = data.set_index('LOCATION')
48data = (data - data.min()) / (data.max() - data.min())
49data = data.groupby(lambda x: x[:3], axis=0).sum()
50
51# Model fitting and evaluation
52X = data[['Value_consumption']]
53y = data['Value_gdp']
54X = sm.add_constant(X)
55
56model = sm.OLS(y, X).fit()
57
58summary = model.summary()
